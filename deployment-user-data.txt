#!/bin/bash
# Update system
apt update && apt upgrade -y

# Install Node.js
curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
apt-get install -y nodejs

# Install PM2
npm install -g pm2

# Create app directory
mkdir -p /home/ubuntu/vauntico-backend
cd /home/ubuntu/vauntico-backend

# Create basic Express server
cat > server.js << 'EOF'
const express = require('express');
const app = express();
const PORT = process.env.PORT || 3000;

app.use(express.json());

// Health check endpoint
app.get('/health', (req, res) => {
  res.json({
    status: 'healthy',
    timestamp: new Date().toISOString(),
    version: '2.0.0',
    uptime: process.uptime(),
    environment: process.env.NODE_ENV || 'production'
  });
});

// Main endpoint
app.get('/', (req, res) => {
  res.json({
    message: 'Vauntico Backend API v2.0',
    status: 'running',
    timestamp: new Date().toISOString(),
    version: '2.0.0'
  });
});

// API status endpoint
app.get('/api/v1/status', (req, res) => {
  res.json({
    status: 'ok',
    version: '2.0.0',
    service: 'vauntico-backend',
    uptime: process.uptime(),
    environment: process.env.NODE_ENV || 'production',
    timestamp: new Date().toISOString()
  });
});

// Start server
app.listen(PORT, '0.0.0.0', () => {
  console.log(`Server running on port ${PORT}`);
});
EOF

# Install dependencies
npm init -y
npm install express

# Start with PM2
pm2 start server.js --name vauntico-backend
pm2 save
pm2 startup
